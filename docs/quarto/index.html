自动生成目录的程序

Code 
rm(list=ls())
library(tidyverse)
library(lubridate)

files = list.files()

map(
  files,
  function(file_name){
    if(grepl(".qmd", file_name, fixed=TRUE)){
      return(
        file.info(file_name) %>%
          rownames_to_column(var = "file_name") %>%
          as_tibble() %>%
          select(file_name, mtime)
        )
    }else{
      return(data.frame())
    }
  }
) %>% 
  bind_rows() %>%
  mutate(
    file_name = str_c("[", file_name,"]", "(", file_name, ")"),
  ) %>%
  arrange(desc(mtime)) %>%
  mutate(
    mtime = as_date(mtime)
  ) %>%
  rename(
    `文件名` = file_name,
    `修改日期` = mtime
  ) -> file_info_df
Note

使用rownames_to_column(var = “file_name”)将df的行名转化为一列，便于转化为tibble

结果已经按照最后保存日期排序
