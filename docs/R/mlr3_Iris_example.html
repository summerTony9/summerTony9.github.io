library(mlr3)
library(tidyverse)
library(mlr3learners)       # load recommended learners provided by mlr3learners package  # this loads further less-well-supported learners
library(mlr3cluster)
library(mlr3extralearners)
data = read_csv("mushrooms.csv", show_col_types = FALSE)
colnames(data) = make.names(colnames(data), unique = T)
data %>%
  mutate_all(as_factor) -> data
```

task = as_task_classif(data, target = "class", id="mushromm_task")
